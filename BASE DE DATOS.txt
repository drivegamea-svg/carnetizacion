### **Solución sin PostGIS (Optimizada para 4 sucursales)**
```sql
-- 1. Crear tabla SIMPLIFICADA (sin PostGIS)
CREATE TABLE puntos_atencion (
    id SERIAL PRIMARY KEY,
    nombre_oficina VARCHAR(100) NOT NULL,
    direccion_publica TEXT NOT NULL,
    latitud FLOAT NOT NULL,   -- Ej: -16.516556
    longitud FLOAT NOT NULL,  -- Ej: -68.221028
    tipo_punto VARCHAR(50) NOT NULL,
    horario_atencion TEXT,
    responsable VARCHAR(100),
    activo BOOLEAN DEFAULT TRUE  -- Eliminación lógica
);

-- 2. Insertar tus 4 sucursales
INSERT INTO puntos_atencion (
    nombre_oficina, direccion_publica, latitud, longitud, tipo_punto, horario_atencion
) VALUES 
    ('JACHA UTHA - Oficina Central', 'Av. Costanera 5022', -16.516556, -68.221028, 'Central', 'L-V 8:00-16:00'),
    ('Terminal Metropolitana', 'Av. América #456', -16.520108, -68.176710, 'Terminal', 'L-D 6:00-22:00'),
    ('Terminal Interprovincial', 'Calle Jose Arzabe', -16.487871, -68.200172, 'Terminal', 'L-D 5:30-23:00'),
    ('Aeropuerto', 'Av. Heroes Km7', -16.509485, -68.168214, 'Aeropuerto', '24/7');

-- 3. Consulta para calcular distancia aproximada (en km) desde un punto
SELECT 
    nombre_oficina,
    111.111 * SQRT(
        POWER(latitud - (-16.50), 2) + 
        POWER((longitud - (-68.18)) * COS(RADIANS(-16.50)), 2)
    ) AS distancia_km
FROM puntos_atencion
WHERE activo = TRUE;
```
